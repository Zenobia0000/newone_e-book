## 第一步：資料結構

```python
import json
import datetime
from typing import Dict, List

class 記帳本:
    def __init__(self):
        self.帳目 = []
        self.類別 = [
            "飲食", "交通", "購物", "娛樂",
            "居住", "醫療", "教育", "其他"
        ]
        self.每月預算 = 30000
        self.載入資料()

    def 載入資料(self):
        """從檔案讀取之前的記錄"""
        try:
            with open("記帳資料.json", "r", encoding="utf-8") as f:
                資料 = json.load(f)
                self.帳目 = 資料.get("帳目", [])
                self.每月預算 = 資料.get("預算", 30000)
        except FileNotFoundError:
            print("第一次使用，歡迎！")

    def 儲存資料(self):
        """將資料存到檔案"""
        資料 = {
            "帳目": self.帳目,
            "預算": self.每月預算
        }
        with open("記帳資料.json", "w", encoding="utf-8") as f:
            json.dump(資料, f, ensure_ascii=False, indent=2)
```

「為什麼要儲存資料？」小美問。

持久化——讓資料在程式關閉後還能保存，這是真實專案的必備功能。

def 新增支出(self, 金額: float, 類別: str, 備註: str = ""):
    """記錄一筆支出"""
    if 類別 not in self.類別:
        print(f"類別 '{類別}' 不存在，已歸類為『其他』")
        類別 = "其他"

    帳目 = {
        "日期": datetime.datetime.now().strftime("%Y-%m-%d"),
        "時間": datetime.datetime.now().strftime("%H:%M"),
        "類型": "支出",
        "金額": 金額,
        "類別": 類別,
        "備註": 備註
    }

    self.帳目.append(帳目)
    self.儲存資料()

    # 檢查預算
    self.檢查預算()

    print(f"✓ 已記錄：{類別} -${金額}")

def 新增收入(self, 金額: float, 來源: str = "薪資"):
    """記錄一筆收入"""
    帳目 = {
        "日期": datetime.datetime.now().strftime("%Y-%m-%d"),
        "時間": datetime.datetime.now().strftime("%H:%M"),
        "類型": "收入",
        "金額": 金額,
        "類別": 來源,
        "備註": ""
    }

    self.帳目.append(帳目)
    self.儲存資料()

    print(f"✓ 已記錄：{來源} +${金額}")

## 統計分析：讓數據說話

```python
def 本月統計(self):
    """顯示本月收支統計"""
    本月 = datetime.datetime.now().strftime("%Y-%m")

    本月帳目 = [
        帳目 for 帳目 in self.帳目
        if 帳目["日期"].startswith(本月)
    ]

    總收入 = sum(
        帳目["金額"] for 帳目 in 本月帳目
        if 帳目["類型"] == "收入"
    )

    總支出 = sum(
        帳目["金額"] for 帳目 in 本月帳目
        if 帳目["類型"] == "支出"
    )

    餘額 = 總收入 - 總支出

    print(f"\n📊 {本月} 月統計")
    print("=" * 40)
    print(f"總收入：${總收入:,.0f}")
    print(f"總支出：${總支出:,.0f}")
    print(f"餘額：  ${餘額:,.0f}")

    # 支出分類統計
    分類統計 = {}
    for 帳目 in 本月帳目:
        if 帳目["類型"] == "支出":
            類別 = 帳目["類別"]
            分類統計[類別] = 分類統計.get(類別, 0) + 帳目["金額"]

    if 分類統計:
        print("\n📈 支出分類：")
        for 類別, 金額 in sorted(分類統計.items(), key=lambda x: x[1], reverse=True):
            百分比 = (金額 / 總支出) * 100
            進度條 = "■" * int(百分比 / 5)
            print(f"  {類別:　<4} ${金額:>8,.0f} {進度條} {百分比:.1f}%")
```

## 預算管理：自我約束

```python
def 設定預算(self, 金額: float):
    """設定每月預算"""
    self.每月預算 = 金額
    self.儲存資料()
    print(f"✓ 每月預算已設定為：${金額:,.0f}")

def 檢查預算(self):
    """檢查是否超出預算"""
    本月 = datetime.datetime.now().strftime("%Y-%m")

    本月支出 = sum(
        帳目["金額"] for 帳目 in self.帳目
        if 帳目["日期"].startswith(本月) and 帳目["類型"] == "支出"
    )

    使用率 = (本月支出 / self.每月預算) * 100
    剩餘預算 = self.每月預算 - 本月支出

    if 使用率 > 100:
        print(f"⚠️ 警告：本月已超支 ${-剩餘預算:,.0f}！")
    elif 使用率 > 80:
        print(f"⚠️ 注意：預算已使用 {使用率:.0f}%，剩餘 ${剩餘預算:,.0f}")
    elif 使用率 > 50:
        print(f"💰 預算使用 {使用率:.0f}%，剩餘 ${剩餘預算:,.0f}")
```

```python
def 刪除最後一筆(self):
    """刪除最後一筆記錄"""
    if not self.帳目:
        print("沒有記錄可以刪除")
        return

    最後一筆 = self.帳目.pop()
    self.儲存資料()

    print(f"✓ 已刪除：{最後一筆['日期']} {最後一筆['類別']} ${最後一筆['金額']}")

def 搜尋帳目(self, 關鍵字: str):
    """搜尋特定帳目"""
    結果 = [
        帳目 for 帳目 in self.帳目
        if 關鍵字 in 帳目.get("備註", "") or 關鍵字 in 帳目["類別"]
    ]

    if 結果:
        for 帳目 in 結果:
            print(f"{帳目['日期']} {帳目['類別']} ${帳目['金額']} {帳目.get('備註', '')}")
    else:
        print(f"找不到包含 '{關鍵字}' 的記錄")
```

每個新功能都讓專案更完整，也讓你的程式能力更進步。
